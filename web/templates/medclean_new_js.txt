function showToast(msg, type = 'info', duration = 4000) {
  const colors = { info:'var(--accent-cyan)', success:'var(--accent-teal)', warn:'var(--accent-amber)', error:'var(--accent-red)' };
  const icons  = { info:'ℹ️', success:'✅', warn:'⚠️', error:'❌' };
  const t = document.createElement('div');
  t.style.cssText = `background:var(--glass-bg);backdrop-filter:blur(20px);border:1px solid ${colors[type]};border-radius:12px;padding:12px 16px;font-family:'Syne',sans-serif;font-size:13px;color:var(--text-primary);display:flex;align-items:center;gap:10px;box-shadow:0 4px 24px rgba(0,0,0,0.3);animation:slideIn 0.3s ease;cursor:pointer`;
  t.innerHTML = `<span>${icons[type]}</span><span style="flex:1">${msg}</span>`;
  t.onclick = () => t.remove();
  const style = document.createElement('style');
  style.textContent = '@keyframes slideIn{from{opacity:0;transform:translateX(30px)}to{opacity:1;transform:translateX(0)}}';
  document.head.appendChild(style);
  toastContainer.appendChild(t);
  setTimeout(() => { t.style.opacity='0'; t.style.transition='opacity 0.3s'; setTimeout(() => t.remove(), 300); }, duration);
}

async function checkBackend() {

      backendStatusEl.style.color = 'var(--accent-red)';
    }
  }
}

// ── THREE.JS HERO SCENE ───────────────────────────────────────────────────────
(function initThreeScene() {
  if (prefersReducedMotion || typeof THREE === 'undefined') return;
  const canvas = document.getElementById('hero-canvas');
  const renderer = new THREE.WebGLRenderer({ canvas, antialias: false, alpha: true });
  renderer.setPixelRatio(Math.min(window.devicePixelRatio, 2));
  renderer.setSize(window.innerWidth, window.innerHeight);
  renderer.setClearColor(0x000000, 0);
  const scene = new THREE.Scene();
  const camera = new THREE.PerspectiveCamera(60, window.innerWidth / window.innerHeight, 0.1, 100);
  camera.position.set(0, 0, 10);
  const particleCount = isMobile ? 1000 : 2000;
  const nodeCount = isMobile ? 40 : 80;
  const mouse = { x: 0, y: 0, tx: 0, ty: 0 };
  let scrollY = 0, frameTime = 0;
  // DNA Helix
  const helixGeo = new THREE.BufferGeometry();
  const helixPos = new Float32Array(particleCount * 3);
  const helixCol = new Float32Array(particleCount * 3);
  for (let i = 0; i < particleCount; i++) {
    const t = (i / particleCount) * Math.PI * 8, strand = i % 2 === 0 ? 1 : -1;
    helixPos[i*3] = Math.cos(t)*strand*1.2; helixPos[i*3+1] = t/(Math.PI*4)*10-5; helixPos[i*3+2] = Math.sin(t)*strand*1.2;
    const mix = i/particleCount; helixCol[i*3]=mix; helixCol[i*3+1]=1-mix*0.2; helixCol[i*3+2]=1;
  }
  helixGeo.setAttribute('position', new THREE.BufferAttribute(helixPos.slice(), 3));
  helixGeo.setAttribute('color', new THREE.BufferAttribute(helixCol, 3));
  const helix = new THREE.Points(helixGeo, new THREE.PointsMaterial({ size:0.04,vertexColors:true,transparent:true,opacity:0.85,sizeAttenuation:true }));
  helix.position.set(3, 0, -2); scene.add(helix);
  // Nodes
  const nodeGeo = new THREE.SphereGeometry(0.08, 6, 6);
  const nodes = [], nodePositions = [];
  for (let i = 0; i < nodeCount; i++) {
    const mesh = new THREE.Mesh(nodeGeo, new THREE.MeshBasicMaterial({ color: i < 3 ? 0xffb800 : 0x00d4ff, transparent:true, opacity: i < 3 ? 0.9 : 0.7 }));
    const pos = new THREE.Vector3((Math.random()-0.5)*18,(Math.random()-0.5)*10,(Math.random()-0.5)*8-4);
    mesh.position.copy(pos); if (i < 3) mesh.scale.setScalar(3.5);
    mesh.userData = { phase: Math.random()*Math.PI*2, speed: 0.5+Math.random()*0.5 };
    scene.add(mesh); nodes.push(mesh); nodePositions.push(pos.clone());
  }
  // Edge network
  const lineVerts = [];
  for (let i = 0; i < nodeCount; i++) for (let j = i+1; j < nodeCount; j++)
    if (nodePositions[i].distanceTo(nodePositions[j]) < 4) { lineVerts.push(nodePositions[i].x,nodePositions[i].y,nodePositions[i].z,nodePositions[j].x,nodePositions[j].y,nodePositions[j].z); }
  const lineGeo = new THREE.BufferGeometry(); lineGeo.setAttribute('position', new THREE.Float32BufferAttribute(lineVerts, 3));
  scene.add(new THREE.LineSegments(lineGeo, new THREE.LineBasicMaterial({ color:0x00d4ff,transparent:true,opacity:0.12 })));
  // Grid
  const gridHelper = new THREE.GridHelper(40, 40, 0x00d4ff, 0x00d4ff);
  gridHelper.material.transparent = true; gridHelper.material.opacity = 0.04; gridHelper.position.y = -5; scene.add(gridHelper);
  // Ambient
  const ambCount = isMobile ? 250 : 500, ambGeo = new THREE.BufferGeometry();
  const ambPos = new Float32Array(ambCount*3), ambVel = new Float32Array(ambCount*3);
  for (let i = 0; i < ambCount; i++) {
    ambPos[i*3]=(Math.random()-0.5)*20; ambPos[i*3+1]=(Math.random()-0.5)*12; ambPos[i*3+2]=(Math.random()-0.5)*10-5;
    ambVel[i*3]=(Math.random()-0.5)*0.003; ambVel[i*3+1]=(Math.random()-0.5)*0.003;
  }
  ambGeo.setAttribute('position', new THREE.BufferAttribute(ambPos, 3));
  scene.add(new THREE.Points(ambGeo, new THREE.PointsMaterial({ size:0.025,color:0xaaccff,transparent:true,opacity:0.3 })));
  document.addEventListener('mousemove', e => { mouse.tx=(e.clientX/window.innerWidth-0.5)*2; mouse.ty=-(e.clientY/window.innerHeight-0.5)*2; });
  window.addEventListener('scroll', () => { scrollY = window.scrollY; }, { passive:true });
  window.addEventListener('resize', () => { camera.aspect=window.innerWidth/window.innerHeight; camera.updateProjectionMatrix(); renderer.setSize(window.innerWidth,window.innerHeight); });
  let last = 0;
  function animate(ts) {
    requestAnimationFrame(animate);
    const delta = Math.min((ts-last)/1000, 0.05); last=ts; if (delta < 1/65) return; frameTime += delta;
    mouse.x += (mouse.tx-mouse.x)*0.05; mouse.y += (mouse.ty-mouse.y)*0.05;
    helix.rotation.y += 0.002; helix.position.y = (frameTime*0.3)%10-5; helix.rotation.x = mouse.y*0.25; helix.rotation.z = mouse.x*0.15;
    nodes.forEach((n,i) => { const s=0.9+Math.sin(frameTime*n.userData.speed+n.userData.phase)*0.15; n.scale.setScalar(i<3?s*3.5:s); });
    const aPos = ambGeo.attributes.position.array;
    for (let i = 0; i < ambCount; i++) {
      aPos[i*3] += ambVel[i*3]+(Math.random()-0.5)*0.001; aPos[i*3+1] += ambVel[i*3+1]+(Math.random()-0.5)*0.001;
      if (Math.abs(aPos[i*3]) > 10) aPos[i*3] *= -0.9; if (Math.abs(aPos[i*3+1]) > 6) aPos[i*3+1] *= -0.9;
    }
    ambGeo.attributes.position.needsUpdate = true;
    gridHelper.position.z = (scrollY*0.005)%1;
    camera.position.x = mouse.x*0.3; camera.position.y = mouse.y*0.2; camera.lookAt(scene.position);
    renderer.render(scene, camera);
  }
  animate(0);
  document.getElementById('hero-canvas').classList.add('visible');
})();

// ── PIPELINE NODE STEPPER ─────────────────────────────────────────────────────
const pipeNodes = document.querySelectorAll('.pipe-node');
function resetPipeline() {
  pipeNodes.forEach(n => {
    n.classList.remove('active-node');
    const s = n.querySelector('.pipe-node-status');
    if (s) { s.className = 'pipe-node-status pipe-status-wait'; s.textContent = 'WAITING'; }
  });
}
function activatePipeStage(idx) {
  pipeNodes.forEach((n, i) => {
    const s = n.querySelector('.pipe-node-status');
    if (!s) return;
    if (i < idx) { n.classList.remove('active-node'); s.className = 'pipe-node-status pipe-status-done'; s.textContent = 'DONE'; }
    else if (i === idx) { n.classList.add('active-node'); s.className = 'pipe-node-status pipe-status-active'; s.textContent = 'ACTIVE'; }
    else { n.classList.remove('active-node'); s.className = 'pipe-node-status pipe-status-wait'; s.textContent = 'QUEUED'; }
  });
}

// ── REAL FILE UPLOAD → /api/upload ────────────────────────────────────────────
async function uploadFile(file) {
  if (!file) return;
  const ext = file.name.split('.').pop().toLowerCase();
  if (!['csv','json','xlsx','xml'].includes(ext)) {
    showToast(`Unsupported format: .${ext}. Use CSV, JSON, XLSX, or XML`, 'error');
    return;
  }
  resetPipeline();
  const dropZone = document.getElementById('drop-zone');
  if (dropZone) { dropZone.querySelector('.upload-text').textContent = `⏳ Uploading ${file.name}…`; dropZone.style.borderColor = 'var(--accent-cyan)'; }

  addLog(`[UPLOAD] ${file.name} (${(file.size/1024).toFixed(1)} KB)`, 'info');
  activatePipeStage(0);

  const fd = new FormData();
  fd.append('file', file);
  try {
    showToast(`Uploading ${file.name} to backend…`, 'info', 3000);
    const res = await fetch(`${API}/api/upload`, { method:'POST', body:fd, mode:'cors' });
    const data = await res.json();

    if (!res.ok) { addLog(`[ERROR] Upload failed: ${data.error || res.status}`, 'err'); showToast(`Upload failed: ${data.error || 'Server error'}`, 'error'); return; }

    activeSessionId = data.session_id;
    uploadedRows = data.overview?.rows || 0;
    const cols = data.overview?.columns || 0;
    const missingCells = data.quality_issues?.filter(q => q.type === 'missing_value').length || 0;

    addLog(`[SUCCESS] Session: ${activeSessionId}`, '');
    addLog(`[SCHEMA] Detected ${cols} columns, ${uploadedRows} rows`, 'info');
    activatePipeStage(1);

    await new Promise(r => setTimeout(r, 400));
    addLog(`[ANALYZE] Running quality analysis…`, 'info');
    activatePipeStage(2);

    // Quality score from server data
    const qs = data.quality_scores;
    qualityScore = qs?.overall || 0;
    anomaliesCount = data.quality_issues?.filter(q => q.type === 'outliers').length || 0;
    const dupCount = data.quality_issues?.filter(q => q.type === 'duplicates').reduce((s,q) => s + (parseInt(q.message)||0), 0) || 0;

    await new Promise(r => setTimeout(r, 500));
    addLog(`[QUALITY] Score: ${qualityScore}/100`, qualityScore > 80 ? '' : 'warn');
    addLog(`[ISSUES] Missing cells: ${missingCells} | Anomalies: ${anomaliesCount} | Duplicates: ${dupCount}`, missingCells > 0 ? 'warn' : '');
    activatePipeStage(3);
    await new Promise(r => setTimeout(r, 400));
    activatePipeStage(4);
    addLog(`[SCHEMA] Column mapping complete`, '');
    await new Promise(r => setTimeout(r, 400));
    activatePipeStage(5);
    addLog(`[VALIDATE] Data validation complete ✓`, '');
    await new Promise(r => setTimeout(r, 300));
    activatePipeStage(6);
    addLog(`[READY] Pipeline complete — data ready for export`, 'info');

    // Update UI
    if (dropZone) { dropZone.querySelector('.upload-text').textContent = `✓ ${file.name} — ${uploadedRows} rows`; dropZone.style.borderColor = 'var(--accent-teal)'; }

    updateKPIs(data);
    updateCharts(data);
    renderDataTable(data, file.name);
    showToast(`✅ ${file.name} processed — ${uploadedRows} rows, quality: ${qualityScore}%`, 'success', 6000);

  } catch (err) {
    addLog(`[ERROR] ${err.message}`, 'err');
    showToast(`Upload error: ${err.message}. Is the Flask server running?`, 'error', 8000);
    if (dropZone) { dropZone.querySelector('.upload-text').textContent = 'Drop Medical Dataset Here'; dropZone.style.borderColor = ''; }
  }
}

// ── UPDATE KPI CARDS FROM REAL DATA ──────────────────────────────────────────
function updateKPIs(data) {
  const recEl = document.getElementById('kpi-records');
  if (recEl) { let n = 0; const iv = setInterval(() => { n = Math.min(n + Math.ceil(uploadedRows/40), uploadedRows); recEl.textContent = n.toLocaleString(); if (n >= uploadedRows) clearInterval(iv); }, 30); }

  const circle = document.getElementById('quality-circle');
  const kpiQ = document.getElementById('kpi-quality');
  let prog = 0;
  const iv2 = setInterval(() => {
    prog = Math.min(prog + 1.2, qualityScore);
    if (circle) circle.style.strokeDashoffset = (125.6 - 125.6 * prog / 100).toFixed(2);
    if (kpiQ) kpiQ.textContent = prog.toFixed(1) + '%';
    if (prog >= qualityScore) clearInterval(iv2);
  }, 20);

  const anomEl = document.getElementById('kpi-anomalies');
  if (anomEl) { let n = 0; const iv3 = setInterval(() => { n = Math.min(n+1, anomaliesCount); anomEl.textContent = n; if (n >= anomaliesCount) clearInterval(iv3); }, 50); }

  const pipEl = document.getElementById('kpi-pipelines');
  if (pipEl) pipEl.textContent = '1';
}

// ── UPDATE CHARTS FROM REAL DATA ─────────────────────────────────────────────
function updateCharts(data) {
  const qs = data.quality_scores || {};
  const issues = data.quality_issues || [];
  const days = Array.from({length:20}, (_,i) => { const d=new Date(); d.setDate(d.getDate()-(19-i)); return d.toLocaleDateString('en-US',{month:'short',day:'numeric'}); });

  // Fabricate a trend converging to the real overall score
  const start = Math.max(20, (qs.overall||70) - 30);
  const qualityData = days.map((_,i) => +(start + (qs.overall-start)*(i/19) + (Math.random()-0.5)*3).toFixed(1));
  qualityData[qualityData.length-1] = qs.overall || qualityData[qualityData.length-1];

  const ctxLine = document.getElementById('chart-line');
  const ctxDonut = document.getElementById('chart-doughnut');
  if (lineChartInstance) lineChartInstance.destroy();
  if (doughnutChartInstance) doughnutChartInstance.destroy();

  const chartOpts = (type) => ({
    responsive:true, maintainAspectRatio:false,
    animation:{ duration:900, easing:'easeInOutQuart' },
    plugins:{ legend:{ display: type==='doughnut', position:'right', labels:{ color:'#4a7fa5', font:{ family:'DM Mono', size:10 }, padding:12 } }, tooltip:{ backgroundColor:'#060f1e', borderColor:'rgba(0,212,255,0.3)', borderWidth:1, titleColor:'#e8f4fd', bodyColor:'#4a7fa5', bodyFont:{ family:'DM Mono' } } },
    scales: type==='line' ? {
      x:{ ticks:{ color:'#4a7fa5', font:{ family:'DM Mono', size:10 }, maxTicksLimit:6 }, grid:{ color:'rgba(0,212,255,0.06)' } },
      y:{ ticks:{ color:'#4a7fa5', font:{ family:'DM Mono', size:10 } }, grid:{ color:'rgba(0,212,255,0.06)' }, min:0, max:100 }
    } : {}
  });

  if (ctxLine) lineChartInstance = new Chart(ctxLine, { type:'line', data:{ labels:days, datasets:[{ label:'Quality Score', data:qualityData, borderColor:'#00d4ff', backgroundColor:'rgba(0,212,255,0.05)', fill:true, tension:0.4, pointRadius:0, pointHoverRadius:4, borderWidth:2 }] }, options:chartOpts('line') });

  // Real issue distribution from backend
  const missing = issues.filter(i=>i.type==='missing_value').length;
  const dups = issues.filter(i=>i.type==='duplicates').length;
  const outliers = issues.filter(i=>i.type==='outliers').length;
  const mixed = issues.filter(i=>i.type==='mixed_types').length;
  const dates = issues.filter(i=>i.type==='date_strings').length;
  const total = missing+dups+outliers+mixed+dates || 1;
  const donutData = [missing, dups, outliers, mixed, dates].map(v => +(v/total*100).toFixed(1));

  if (ctxDonut) doughnutChartInstance = new Chart(ctxDonut, { type:'doughnut', data:{ labels:['Missing Values','Duplicates','Outliers','Mixed Types','Date Issues'], datasets:[{ data:donutData, backgroundColor:['rgba(0,212,255,0.8)','rgba(0,255,200,0.8)','rgba(255,184,0,0.8)','rgba(255,59,107,0.8)','rgba(123,94,167,0.8)'], borderColor:'#060f1e', borderWidth:3, hoverOffset:8 }] }, options:chartOpts('doughnut') });
}

// ── RENDER REAL DATA TABLE ────────────────────────────────────────────────────
function renderDataTable(data, fileName) {
  const emptyEl = document.getElementById('table-empty-state');
  const wrapper = document.getElementById('table-wrapper');
  const thead = document.getElementById('table-head');
  const tbody = document.getElementById('table-body');
  const titleEl = document.getElementById('table-title-label');
  if (!tbody || !thead) return;

  const rows = (data.preview?.rows || []).slice(0, 20);
  if (!rows.length) { if (emptyEl) emptyEl.style.display='block'; if (wrapper) wrapper.style.display='none'; return; }

  const cols = Object.keys(rows[0]);
  if (titleEl) titleEl.textContent = `${fileName} — ${data.overview?.rows||0} rows, ${cols.length} columns`;

  thead.innerHTML = '<tr>' + cols.map(c => `<th>${c.replace(/_/g,' ')}</th>`).join('') + '</tr>';
  tbody.innerHTML = rows.map(row => {
    const cells = cols.map(c => {
      const val = row[c];
      const isEmpty = val === null || val === undefined || val === '';
      return `<td>${isEmpty ? '<span style="color:var(--text-muted);font-style:italic">[empty]</span>' : String(val).substring(0,60)}</td>`;
    }).join('');
    return `<tr>${cells}</tr>`;
  }).join('');

  if (emptyEl) emptyEl.style.display = 'none';
  if (wrapper) wrapper.style.display = 'block';
}
